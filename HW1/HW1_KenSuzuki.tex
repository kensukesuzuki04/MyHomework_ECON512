\begin{align*}
\vect{Y1} = \left[\begin{array}{cccccccc} -\frac{3}{2} & -\frac{5}{4} & -\frac{1}{2} & 0 & \frac{1}{2} & \frac{3}{2} & \frac{5}{2} & 3 \end{array}\right]\\
\vect{Y2} = \left[\begin{array}{cccccccc} -\frac{3}{2} & -\frac{7}{8} & \frac{5}{2} & 6 & \frac{21}{2} & \frac{45}{2} & \frac{77}{2} & 48 \end{array}\right]
\end{align*}
